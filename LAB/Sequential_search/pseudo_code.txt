#include <stdio.h>
#定義一個 SIZE 為 5

定義函式 search (宣告參數 arr[], target, size 為整數){

    宣告 i 為整數;
    for 初始化 i 為 0 且從 0 持續遞增到 size {
        如果在 arr 陣列中的第 i 個值等於要找的值(target){
            找到符合條件時回傳索引值 +1;
        }
    }
    未找到符合條件時回傳 -1;
} 
int main(){

    宣告一個整數陣列 arr 大小為 SIZE;
    宣告四個整數 i, target, element, *ptr;
    將 ptr 指標到陣列 arr;
    =======  循序搜尋法  =======換行; 
    ----  請連續輸入五個數值 ----換行; 
    換行;
    for 初始化 i 為 0 且從 0 持續遞增到 SIZE {
        請輸入第 i + 1 個值 : ;
        輸入整數 ptr + i 依序存入 arr 陣列元素中;
    }
    換行;
    請輸入搜尋的數值 : ;
    輸入搜尋的數值(target);
    呼叫函式 search 對arr、搜尋的值(target)、數量(SIZE) 進行搜尋並將結果傳給變數 element;
    如果 element 不等於 -1 {
        target 是第 element 個數字！;
    }
    否則 {
        沒有 target 這個數字;
    }

    return 0;
}
